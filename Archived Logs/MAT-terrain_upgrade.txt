### Terrain Classification Algorithm

```javascript
async function classifyTerrain(centerLat, centerLon, radiusNm = 5) {
  // Sample 9 points in a grid around the center
  const samples = await getElevationSamples(centerLat, centerLon, radiusNm);
  
  const elevations = samples.map(s => s.elevation);
  const min = Math.min(...elevations);
  const max = Math.max(...elevations);
  const range = max - min;
  const stdDev = calculateStdDev(elevations);
  
  // Classify based on elevation variability
  if (range < 500 && stdDev < 100) return 'flat';        // Plains, desert
  if (range < 1500 && stdDev < 300) return 'moderate';   // Foothills
  return 'mountainous';                                    // Canyons, alpine
}
```
